from tkinter import * from tkinter.colorchooser import * from tkinter.filedialog import * from tkinter.messagebox import *

class Application(Frame): def init(self, master=None): super().init(master) self.master = master self.name_file = None self.context_menu = None self.create_widget()

def create_widget(self):
    """创建主菜单"""
    menu_bar = Menu(root)

    # 创建子菜单
    menu_file = Menu(menu_bar)
    menu_edit = Menu(menu_bar)
    menu_help = Menu(menu_bar)

    # 将子菜单加入到菜单栏
    menu_bar.add_cascade(label="文件(F)", menu=menu_file)
    menu_bar.add_cascade(label="编辑(E)", menu=menu_edit)
    menu_bar.add_cascade(label="帮助(H)", menu=menu_help)

    # 增加菜单项
    menu_file.add_command(label="新建", accelerator="ctrl+n", command=self.file_new)
    menu_file.add_command(label="打开", accelerator="ctrl+o", command=self.file_open)
    menu_file.add_command(label="保存", accelerator="ctrl+s", command=self.file_save)
    menu_file.add_separator()
    menu_file.add_command(label="退出", accelerator="ctrl+q",command=self.exit)

    # 增加快捷键
    root.bind("<Control-n>", lambda event:self.file_new())
    root.bind("<Control-o>", lambda event:self.file_open())
    root.bind("<Control-s>", lambda event: self.file_save())
    root.bind("<Control-q>", lambda event: self.exit())

    # 主窗口绑定菜单
    root["menu"] = menu_bar

    # 创建文本编辑区
    self.text_pad = Text(root, width=50, height=30)
    self.text_pad.pack()

    # 右键绑定上下文菜单事件
    root.bind("<Button-3>", self.create_context_menu)

    # 创建上下文菜单
    self.context_menu = Menu(root)
    self.context_menu.add_command(label="背景颜色", command=self.openAskColor)

def file_new(self):
    self.name_file = asksaveasfilename(title="另存为", initialfile="未命名.txt", defaultextension=".txt",
                      filetypes=[("文档文件", "*.txt")])
    self.file_save()

def file_open(self):
    self.text_pad.delete(1.0, END)
    with askopenfile(title="打开文件") as f:
        self.text_pad.insert(INSERT, f.read())
        self.name_file = f.name

def file_save(self):
    if self.name_file:
        with open(self.name_file, "w") as f:
            c = self.text_pad.get(1.0, END)
            f.write(c)
    else:
        self.file_new()

def exit(self):
    root.quit()

def openAskColor(self):
    s1 = askcolor(color="red", title="背景颜色")
    root["bg"] = s1[1]

def create_context_menu(self,event):
    """上下文菜单在鼠标点击出的坐标显示"""
    self.context_menu.post(event.x_root, event.y_root)

if name == “main": root = Tk() root.geometry(“500x300”) app = Application(root) root.title(“我的记事本”) root.mainloop()
